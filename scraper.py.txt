name: Daily Digikala Scraper

on:
  schedule:
    - cron: '0 6 * * *'  # هر روز ساعت 06:00 UTC
  workflow_dispatch:

jobs:
  run-scraper:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Run Digikala Scraper
        run: |
          python scraper.py --items 200 --pages 30 --out digikala_top200_$(date +%F).csv

      - name: Upload CSV
        uses: actions/upload-artifact@v4
        with:
          name: digikala-top200-$(date +%F)
          path: digikala_top200_$(date +%F).csv
